📚 Quick Start

Get started with DOM Helpers in seconds

CDN Installation (Recommended)

<!-- Latest version --> <script src="https://cdn.jsdelivr.net/npm/dom-helpers@latest/dist/dom-helpers.min.js"></script> <!-- Start using immediately --> <script> // Elements Helper Elements.myButton.update({ textContent: "Click me!", style: { backgroundColor: "blue" } }); // Collections Helper Collections.ClassName('card').update({ style: { boxShadow: "0 4px 8px rgba(0,0,0,0.1)" } }); // Selector Helper Selector.queryAll('.btn').update({ classList: { add: ["enhanced"] } }); </script>

NPM Installation

# Install via npm npm install dom-helpers # ES6 Modules import { Elements, Collections, Selector } from 'dom-helpers'; # CommonJS const { Elements, Collections, Selector } = require('dom-helpers');
